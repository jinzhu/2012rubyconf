What IS DSL?

用于解决特殊领域中特殊问题表示技术和解决方案的程序设计语言

DSL FOR DSL解决方案

                    class Gemfile::Configuration
  include Qor::Dsl
  default_configs [ENV['BUNDLE_GEMFILE'], 'Gemfile']

  node :gem

  [:git, :path, :group].map do |name|
    node name do
      node :gem
    end
  end
end

查询数据

                        Gemfile::Configuration.find('gem', 'rails').name
#=> rails

# 取出Rails的配置
Gemfile::Configuration.find('gem', 'rails').options
#=> {:git=>"git://github.com/rails/rails.git"}

# 取出所有Development环境下的Gem
development_gems = Gemfile::Configuration.deep_find('gem') do |n|
  parent = n.parent
  parent.root? || !parent.is_node?(:group) || parent.is_node?(:group, :development)
end
development_gems.map(&:name)
#=> ["rails", "capistrano-confirm", "paygent", "qor_dsl", "activesupport", "actionpack", "pry-rails"]

DSL FOR DSL解决方案

                        class Nagios::Configuration
  include Qor::Dsl
  node :contact_emails

  node :mail do
    node :address
  end

  node :server do
    node :contact_emails, inherit: true
    node :host
    node :port, default_value: 22
    node :username, default_value: 'app'

    node :check_disk, default_options: {warning: '10%', critical: '8%'}
    node :check_load
    node :check_memcached
    node :check_mysql_slave
  end
end

查询数据

                        # 继承父级
Nagios::Configuration.find('server', 'server1').first(:contact_emails).value
#=> "all_teams@example.com"
# 重写继承
Nagios::Configuration.find('server', 'server2').first(:contact_emails).value
#=> team2@example.com

# 默认
Nagios::Configuration.find('server', 'server1').first(:check_disk).value
#=> {warning: '10%', critical: '8%'}
